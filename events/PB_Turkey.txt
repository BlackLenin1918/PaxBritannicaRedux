country_event = {		#The Civil War begins
	id = turkey.7
	title = turkey.7.t
	desc = turkey.7.d
	picture = GFX_report_event_armored_car

	is_triggered_only = yes

	fire_only_once = yes

	option = {
		name = turkey.7.a

		add_political_power = 50

		TUR = {
			declare_war_on = {
				target = NTR
				type = annex_everything
			}
		}

		NTR = {
			declare_war_on = {
				target = TUR
				type = annex_everything
			}
		}

		hidden_effect = {
			TUR = {
				set_country_flag = TUR_Turkish_Civil_War_Event_Visible
				set_country_flag = Super_Event_Visible
			}

			sound_effect = "super_turkish_war"
		}
	}
}

country_event = {		#End of the Turkish Civil War - Nationalist Victory
	id = turkey.8
	title = turkey.8.t
	desc = turkey.8.d
	picture = GFX_report_event_funeral

	trigger = {
		tag = NTR
		has_global_flag = TUR_Turkish_Civil_War_Fired
		NOT = {
			country_exists = TUR 
			has_war_with = TUR
		}
	}

	fire_only_once = yes

	option = {
		name = turkey.8.a

		add_political_power = -150

		remove_ideas = PB_TUR_Turkish_Civil_War

		add_ideas = PB_TUR_Turkish_Civil_War_Recovery

		hidden_effect = {
			ROOT = {
				drop_cosmetic_tag = yes
			}
		}
	}
}
