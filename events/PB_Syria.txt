		}

		hidden_effect = {
			set_global_flag = syria_england

				transfer_state = 677

			}

				transfer_state = 553
			}

				transfer_state = 781
			}

			load_OOB = LEB_1936
		}
	}

		name = syria.10.a
		ENG = {
			puppet = SYR
			puppet = LEB
		}
		hidden_effect = {
		TUR = { country_event = { id = syria.12 }}}
	}

	option = {
		name = syria.10.b
		ai_chance = { factor = 0 }
	}
}
