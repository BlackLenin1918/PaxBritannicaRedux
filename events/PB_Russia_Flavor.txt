#coronation of anastasia = differing text for absolute/const. 
news_event = {
	id = russia_fl.14
	title = russia_fl.14.t
	desc = {
		text = russia_fl.14.d1
		trigger = {
			has_completed_focus = RUS_absolutism
		}
	}	
	desc = {
		text = russia_fl.14.d2
		trigger = {
			has_completed_focus = RUS_constitutionalism
		}
	}	

	is_triggered_only = yes

	option = { #for russia in either case
		name = russia_fl.14.a
		trigger = {
			SOV = {
				has_completed_focus = RUS_constitutionalism
			}
			tag = SOV
		}
	}

	option = { #for england and allies if russia const
		name = russia_fl.14.b
		trigger = {
			SOV = {
				has_completed_focus = RUS_constitutionalism
			}
			OR = {
				tag = ENG
				is_puppet_of = ENG
				is_in_faction_with = ENG
			}
		}
	}

	option = { #for england and allies if russia absol
		name = russia_fl.14.c
		trigger = {
			SOV = {
				has_completed_focus = RUS_absolutism
			}
			OR = {
				tag = ENG
				is_puppet_of = ENG
				is_in_faction_with = ENG
			}
		}
	}

	option = { #For Austria and allies if russia absol
		name = russia_fl.14.e
		trigger = {
			SOV = {
				has_completed_focus = RUS_absolutism
			}
			OR = {
				tag = AUS
				is_puppet_of = AUS
				is_in_faction_with = AUS
			}
		}
	}

	option = { #for socialist countries (not aligned with eng or aus) in any case
		name = russia_fl.14.f
		trigger = {
			NOT = {
				tag = ENG
				is_puppet_of = ENG
				is_in_faction_with = ENG
				tag = AUS
				is_puppet_of = AUS
				is_in_faction_with = AUS
			}
			OR = {
				has_government = socialism
				has_government = authsoc
			}
		}
	}

	option = { # all other countries if absol
		name = russia_fl.14.g
		trigger = {		
			SOV = { has_completed_focus = RUS_absolutism }	
			NOT = {
				OR = {
					tag = RUS
					tag = ENG
					is_puppet_of = ENG
					is_in_faction_with = ENG
					tag = AUS
					is_puppet_of = AUS
					is_in_faction_with = AUS
					has_government = authsoc
					has_government = socialism
				}
			}
		}
	}

	option = { #all other countries if const
		name = russia_fl.14.h
		trigger = {		
			SOV = { has_completed_focus = RUS_constitutionalism }	
			NOT = {
				OR = {
					tag = RUS
					tag = ENG
					is_puppet_of = ENG
					is_in_faction_with = ENG
					has_government = authsoc
					has_government = socialism
				}
			}
		}
	}

}
