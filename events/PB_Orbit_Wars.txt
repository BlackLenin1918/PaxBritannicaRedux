country_event = {		#Launch Fail Chance Checker
	id = Orbit.1
	title = Orbit.1.t
	desc = Orbit.1.d
	picture = GFX_report_Generic_Poverty

	is_triggered_only = yes

	option = { 
		name = Orbit.1.a

		if = {
			limit = {
				has_country_flag = OBW_Critical_Failure_Assured
			}
			randomize_temp_variable = temp_OBW_seed

			random_list = {
				seed = temp_OBW_seed
				25 = {
					country_event = {
						id = Orbit.2
						days = 1
					}
				}
				25 = {
					country_event = {
						id = Orbit.3
						days = 1
					}
				}
				25 = {
					country_event = {
						id = Orbit.4
						days = 1
					}
				}
				25 = {
					country_event = {
						id = Orbit.5
						days = 1
					}
				}
			}
		}

		if = {
			limit = {
				has_country_flag = OBW_Critical_Failure_High
			}
			randomize_temp_variable = temp_OBW_seed

			random_list = {
				seed = temp_OBW_seed
				75 = {
					country_event = {
						id = Orbit.2
						days = 1
					}
				}
				25 = {
					country_event = {
						id = Orbit.18
						days = 1
					}
				}
			}
		}

		if = {
			limit = {
				has_country_flag = OBW_Critical_Failure_Moderate
			}
			randomize_temp_variable = temp_OBW_seed

			random_list = {
				seed = temp_OBW_seed
				50 = {
					country_event = {
						id = Orbit.2
						days = 1
					}
				}
				50 = {
					country_event = {
						id = Orbit.18
						days = 1
					}
				}
			}
		}

		if = {
			limit = {
				has_country_flag = OBW_Critical_Failure_Low
			}
			randomize_temp_variable = temp_OBW_seed

			random_list = {
				seed = temp_OBW_seed
				25 = {
					country_event = {
						id = Orbit.2
						days = 1
					}
				}
				75 = {
					country_event = {
						id = Orbit.18
						days = 1
					}
				}
			}
		}

		if = {
			limit = {
				has_country_flag = OBW_Critical_Failure_Very_Low
			}
			randomize_temp_variable = temp_OBW_seed

			random_list = {
				seed = temp_OBW_seed
				10 = {
					country_event = {
						id = Orbit.2
						days = 1
					}
				}
				90 = {
					country_event = {
						id = Orbit.18
						days = 1
					}
				}
			}
		}
		
		PB_OBW_Reset_Fail_Chance = yes
	}
}
