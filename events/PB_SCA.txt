### A Bloody Success ###
country_event = {
	id = SCA.9
	title = SCA9title
	desc = SCA9desc
	picture = GFX_report_PB_Graveyard
 
        trigger = {
        tag = SCA
        has_country_flag = SCA_LAK_WAR
        LAK = {
            OR = {
                exists = no 
                has_capitulated = yes
                NOT = { 
                     owns_state = 800
                     owns_state = 389
                     owns_state = 388
                }           
            }
        }
        }
 
        cancel_if_invalid = yes

        fire_only_once = yes


	option = {
		name = SCA9opta
                clr_country_flag = SCA_LAK_WAR
                set_country_flag = SCA_LAK_WAR_OVER     
                hidden_effect = { 
                      LAK = {
                      	   remove_state_core = 800
                  	   remove_state_core = 389
                           remove_state_core = 388
                       }
                }
		ai_chance = {
			factor = 1
                }
        }
}
