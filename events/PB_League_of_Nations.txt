add_namespace = lon

country_event = {		#League of Nations Baltic Mandate
	id = lon.1
	title = lon.1.t
	desc = lon.1.d
	picture = GFX_report_PB_Military_Tribunal

	trigger = {
		tag = ENG
		has_war_with = BAL
		BAL = {
			has_capitulated = yes
			NOT = {
				has_full_control_of_state = 12
			}
		}
		12 = {
			controller = {
				OR = {
					is_in_faction_with = ENG
					tag = ENG
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	fire_only_once = yes

	option = { #NOT APPLICABLE
		name = lon.1.a

		ai_chance = {
			factor = 80
		}

		RUS = {
			end_puppet = BAL
		}

		BAL = {
			drop_cosmetic_tag = yes

			set_cosmetic_tag = BAL_LON_MANDATE

			set_politics = {
				ruling_party = ds
			}

			create_country_leader = {
				name = "Sir Richard McCreery"
				desc = "PB_NO_DESC"
				picture = "Portrait_BAL_McCreery.dds"
				expire = "1965.1.1"
				ideology = DS_VARIANT_06
			}
		}

		ENG = {
			add_to_faction = BAL
		}

		every_country = {
			limit = {
				has_war_with = BAL
			}
			white_peace = BAL
		}
	}
}
