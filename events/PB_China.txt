country_event = { #ROC Rebels spawn
	id = GenChina.6
	title = GenChina.6.t
	desc = GenChina.6.d
	picture = GFX_report_event_Generic_German_Soldier

	fire_only_once = yes

	trigger = {
		ROC = {
			has_war_with = CHI
		}
		OR = {
			AND = {
				date > 1937.5.1
				CHI = {
					surrender_progress < 5
				}
			}
			AND = {
				date > 1938.2.1
				CHI = {
					surrender_progress < 10
				}
			}
			ROC = {
				surrender_progress > 25
			}
		}

	}

	mean_time_to_happen = {
		days = 5
	}


	option = {
		name = GenChina.6.a

		ai_chance = {
			factor = 100
		}
		PB_ROC_Pseudo_Randomized_Rebels_Spawn = yes
	}
}
