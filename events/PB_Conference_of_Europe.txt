#Implementation of Dexter Plan
country_event = {
	id = COE.3
	title = COE.3.t
	desc = COE.3.d
	picture = GFX_report_Palace_Het_Loo
 
    is_triggered_only = yes
	
    trigger = {
    }

	option = {
		name = COE.3.a
	WLA = { transfer_state = 29 }
	BRT = { transfer_state = 14 }
	BSQ = { transfer_state = 841 }

	17 = { set_demilitarized_zone = yes }
	685 = { set_demilitarized_zone = yes }
	785 = { set_demilitarized_zone = yes }
	18 = { set_demilitarized_zone = yes }

	clr_global_flag = PB_COE_PROPOSAL_DEXTER
	clr_global_flag = PB_COE_TOPIC_FRANCE

	if = { 
	limit = { has_global_flag = PB_COE_LORRAINE_OCCUPATION }
	GER = { transfer_state = 17 }
	17 = { set_demilitarized_zone = no }
	}

	if = { 
	limit = { has_global_flag = PB_COE_ISRAEL_IN_FRANCE }

	FRA = { release = ISR }

	ENG = { add_to_faction = ISR }
	}

	if = { 
	limit = { ISR = { owns_state = 18 } }
	18 = { set_demilitarized_zone = no }
	}
    }
}
