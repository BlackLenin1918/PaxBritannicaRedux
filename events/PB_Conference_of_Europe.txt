#The End of the France Debate
country_event = {
	id = COE.19
	title = COE.19.t
	desc = COE.19.d
	picture = GFX_report_Palace_Het_Loo
 
    is_triggered_only = yes

    fire_only_once = yes
	
    trigger = {
    }

	option = {
		name = COE.19.a
		ai_chance = {
			factor = 1
        }
	WLA = { transfer_state = 29 }

	GER = { 
	annex_country = { target = FRA } 
	set_cosmetic_tag = GER_FRANCIA 
	}
	FRA = {
	every_state = { 
			limit = { is_core_of = FRA }
			add_core_of = GER
		}	
	}

	if = { 
	limit = { has_global_flag = PB_COE_FRANCIA_LARGE_OCT }
    		every_state = {
		limit = { check_variable = { culture_id = 8 } }
		add_core_of = OCT
		}
		release = OCT
	}

	if = { 
	limit = { has_global_flag = PB_COE_FRANCIA_SMALL_OCT }
		OCT = { add_state_core = 557 add_state_core = 31 add_state_core = 22 }
		release = OCT 
	}

	if = { 
	limit = { has_global_flag = PB_COE_FRANCIA_NORMANDY }
    		15 = { add_core_of = NRM }
		NRM = { transfer_state = 15 }
		NRM = { add_ideas = PB_Constitutionally_Enshrined_Neutrality }
		country_event = { id = COEGOV.1 days = 0 }
	}

	if = { 
	limit = { has_global_flag = PB_COE_ISRAEL_IN_FRANCE }

	GER = { release = ISR }

	ENG = { add_to_faction = ISR }
	}

	if = { 
	limit = { has_global_flag = PB_COE_FRANCIA_OCT_TWO_SEAS }
	OCT = { transfer_state = 19 } 
	
	}

	if = { 
	limit = { ISR = { owns_state = 18 } }
	18 = { set_demilitarized_zone = no }
	}
    }
}
