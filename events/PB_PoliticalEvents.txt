country_event = { #Democratic Referendum
	id = political.9
	title = political.9.t
	desc = political.9.d
	picture = GFX_report_event_guard_infantry
	
	is_triggered_only = yes
	
	option = { #Conservatives
		name = political.9.a
		trigger = { 
			NOT = {
				has_government = conservative
			}
			conservative > 0.4
		}

		set_politics = {
			ruling_party = conservative
			elections_allowed = no
		}
	}

	option = { #Social Liberals
		name = political.9.b
