country_event = {
	id = westphalia.1
	title = westphalia.1.t
	desc = westphalia.1.d
	picture = GFX_report_event_militia_infantry
	
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_government = authsoc }
		NOT = { has_government = socialist }
	}
	
	option = {
		name = westphalia.1.o1
		set_country_flag = WSP_volksmilizen_supported
		BAV = {
			add_opinion_modifier = {
				target = ROOT
				modifier = support_for_bavarian_exilants
			}
		}
		add_popularity = {
			ideology = authoritarianism
			popularity = 0.025
		}
		
		ai_chance = { factor = 75 }
	}
	
	option = {
		name = westphalia.1.o2
		set_country_flag = WSP_volksmilizen_opposed
		add_political_power = -30
		add_popularity = {
			ideology = authsoc
			popularity = 0.01
		}
		
		ai_chance = { factor = 25 }
	}
}

	}
}
