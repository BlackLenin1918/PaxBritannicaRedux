on_actions = {

	on_startup = {
		effect = {
			every_country = {
				limit = { 
                    national_dissent = yes
                }
                set_variable = { national_dissent = 0 }
                set_variable = { national_dissent_growth = 0.5 }
                set_variable = { national_dissent_reduction = -0.5 }
                set_variable = { national_dissent_max = 100 }
                set_variable = { national_dissent_min = 0 }
            }
		}

		effect = {
			every_country = {
				limit = { 
                    tag = ENG
                }
				add_to_variable = {
				    var = national_dissent
				    value = 10
				}
            }
		}

		effect = {
			every_country = {
				limit = { 
                    tag = FRA
                }
				add_to_variable = {
				    var = national_dissent
				    value = 30
				}
            }
		}
	}

    on_daily = {
		effect = {
            if = {
				limit = { 
                    NOT = {
                    	has_stability < 0.5
                    }
                }
                
                add_to_variable = { national_dissent = national_dissent_growth }

				clamp_variable = {
				    var = national_dissent
				    min = national_dissent_min
				    max = national_dissent_max
				}
            }
        }

		effect = {
            if = {
				limit = { 
					has_war = yes
                    NOT = {
                    	has_war_support < 0.4
                    }
                }
                
                add_to_variable = { national_dissent = national_dissent_growth }

				clamp_variable = {
				    var = national_dissent
				    min = national_dissent_min
				    max = national_dissent_max
				}

            }
        }

		effect = {
            if = {
				limit = { 
					has_stability > 0.5
                }
                
                add_to_variable = { national_dissent = national_dissent_reduction }

				clamp_variable = {
				    var = national_dissent
				    min = national_dissent_min
				    max = national_dissent_max
				}
            }
        }

		effect = {
            if = {
				limit = { 
					has_war = yes
                    has_war_support > 0.4
                }
                
                add_to_variable = { national_dissent = national_dissent_reduction }

				clamp_variable = {
				    var = national_dissent
				    min = national_dissent_min
				    max = national_dissent_max
				}        
            }
        }

		effect = {
            if = {
				limit = { 
					has_country_flag = National_Dissent_Lowering
                }
                
                add_to_variable = { national_dissent = national_dissent_reduction }

				clamp_variable = {
				    var = national_dissent
				    min = national_dissent_min
				    max = national_dissent_max
				}
            }
        }
   	}
