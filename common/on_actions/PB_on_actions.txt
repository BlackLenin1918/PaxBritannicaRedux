on_actions = {

	on_startup = {
		effect = {
			every_country = {
				limit = { 
                    national_dissent = yes
                }
                set_variable = { national_dissent = 0 }
                set_variable = { national_dissent_growth = 0.5 }
                set_variable = { national_dissent_reduction = -0.5 }
            }
		}

		effect = {
			every_country = {
				limit = { 
                    tag = ENG
                }
				add_to_variable = {
				    var = national_dissent
				    value = 10
				}
            }
		}

		effect = {
			every_country = {
				limit = { 
                    tag = FRA
                }
				add_to_variable = {
				    var = national_dissent
				    value = 30
				}
            }
		}
	}

    on_daily = {
		effect = {
            if = {
				limit = { 
					check_variable = { national_dissent < 100 }
                    NOT = {
                    	has_stability < 0.5
                    }
                }
                
                add_to_variable = { national_dissent = national_dissent_growth }
            }
        }
   	}
}
