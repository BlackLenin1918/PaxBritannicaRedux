	on_monthly_RUS = {
		effect = {
			RUS = {
				if  = { 
					limit = {
						OR = {
						    check_variable = { RUS_Political_Dissonance > 0.74 }
							check_variable = { RUS_Political_Dissonance = 0.74 }
						}
						NOT = { 
							has_idea = RUS_High_Political_Dissonance
						}
					}
					hidden_effect = {
						remove_ideas = {
							RUS_Low_Political_Dissonance
							RUS_Med_Political_Dissonance
						}
						add_ideas = RUS_High_Political_Dissonance
					}
				}
				else_if  = { 
					limit = {
							check_variable = { RUS_Political_Dissonance < 0.74 }

						}
						NOT = { 
							has_idea = RUS_Med_Political_Dissonance
						}
					}

						}
						add_ideas = RUS_Med_Political_Dissonance
					}
				}
				else_if = { 
					limit = {
						OR = {
						    check_variable = { RUS_Political_Dissonance > 0.0 }
							check_variable = { RUS_Political_Dissonance = 0.49 }
						}
						NOT = { 
							has_idea = RUS_Low_Political_Dissonance
						}
					}
					hidden_effect = {
						remove_ideas = {
							RUS_Med_Political_Dissonance
							RUS_High_Political_Dissonance
						}
						add_ideas = RUS_Low_Political_Dissonance
					}
				}
			}
			RUS = {
				if  = { 
					limit = {
						OR = {
						    check_variable = { RUS_Ethnic_Cohesion > 0.74 }
							check_variable = { RUS_Ethnic_Cohesion = 0.74 }
						}
						NOT = { 
							has_idea = RUS_High_Ethnic_Cohesion_Idea
						}
					}

						remove_ideas = {
							RUS_Low_Ethnic_Cohesion_Idea
							RUS_Med_Ethnic_Cohesion_Idea
						}
						add_ideas = RUS_High_Ethnic_Cohesion_Idea
					}
				}
				else_if  = { 
					limit = {
							check_variable = { RUS_Ethnic_Cohesion < 0.74 }
							check_variable = { RUS_Ethnic_Cohesion  > 0.49 }
						}
						NOT = { 
							has_idea = RUS_Med_Ethnic_Cohesion_Idea
						}
					}
					hidden_effect = {
						remove_ideas = {
							RUS_High_Ethnic_Cohesion_Idea

						}
						add_ideas = RUS_Med_Ethnic_Cohesion_Idea

						OR = {
						    check_variable = { RUS_Ethnic_Cohesion > 0.0 }
							check_variable = { RUS_Ethnic_Cohesion = 0.49 }
