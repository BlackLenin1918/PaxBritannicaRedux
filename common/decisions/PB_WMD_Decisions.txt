	PB_Launch_Chemical_Attack = {

		icon = generic_gas_mask

		cost = 0

		ai_will_do = {
			factor = 0
		}

		available = {
			has_war = yes

			OR = {
				custom_trigger_tooltip = {
					tooltip = PB_PB_Has_Developed_Choking_Chemicals_Tooltip
					has_country_flag = PB_Has_Noxious_ChemicalsII
				}		

				custom_trigger_tooltip = {
					tooltip = PB_PB_Has_Developed_Nerve_Chemicals_Tooltip
					has_country_flag = PB_Has_Noxious_ChemicalsIII
				}		
			}

			NOT = {
				OR = {
					has_idea = PB_Choking_Agents_Deployed
					has_idea = PB_Nerve_Agents_Deployed
				}
			}
		}

		modifier = {
		}

		days_remove = 5

		days_re_enable = 180

		remove_effect = {
			if = {
				limit = {
					has_country_flag = PB_Has_Noxious_ChemicalsII
				}
				add_timed_idea = {
					idea = PB_Choking_Agents_Deployed
					days = 60
				}
			}

			if = {
				limit = {
					AND = {
						has_country_flag = PB_Has_Noxious_ChemicalsII
						has_country_flag = PB_Has_Noxious_ChemicalsIII
					}
				}
				add_timed_idea = {
					idea = PB_Nerve_Agents_Deployed
					days = 60
				}
			}				
		}
	}
