PB_RUS_Rasputin_Influence = {

	PB_RUS_Rasputin_Influence_Growth = {

		days_mission_timeout = 21

		icon = generic_decision

		activation = {
			always = no
		}

		selectable_mission = no

		is_good = no

		ai_will_do = {
			factor = 10
		}

		timeout_effect = {
			activate_mission = PB_RUS_Rasputin_Influence
			add_to_variable = { PB_RUS_Rasputin_Influence_var = 0.05 }
			if = {
				limit = {
					check_variable = {
						var = PB_RUS_Rasputin_Influence_var
						value = 0.1
						compare = less_than_or_equals
					}
				}
				add_ideas = PB_RUS_Rasputin_Influence_1
			}
			else_if = {
				limit = {
					AND = {
						check_variable = {
							var = PB_RUS_Rasputin_Influence_var
							value = 0.1
							compare = greater_than
						}
						check_variable = {
							var = PB_RUS_Rasputin_Influence_var
							value = 0.25
							compare = less_than_or_equals
						}
					}
				}
				add_ideas = PB_RUS_Rasputin_Influence_2
			}
			else_if = {
				limit = {
					AND = {
						check_variable = {
							var = PB_RUS_Rasputin_Influence_var
							value = 0.25
							compare = greater_than
						}
						check_variable = {
							var = PB_RUS_Rasputin_Influence_var
							value = 0.5
							compare = less_than_or_equals
						}
					}
				}
				add_ideas = PB_RUS_Rasputin_Influence_3
			}
			else_if = {
				limit = {
					AND = {
						check_variable = {
							var = PB_RUS_Rasputin_Influence_var
							value = 0.5
							compare = greater_than
						}
						check_variable = {
							var = PB_RUS_Rasputin_Influence_var
							value = 0.75
							compare = less_than_or_equals
						}
					}
				}
				add_ideas = PB_RUS_Rasputin_Influence_4
			}
			else_if = {
				limit = {
					check_variable = {
						var = PB_RUS_Rasputin_Influence_var
						value = 0.75
						compare = greater_than
					}
				}
				add_ideas = PB_RUS_Rasputin_Influence_5
			}
			if = {
				limit = {
					has_country_leader = {
						character = RUS_Xenia_Olga
						ruling_only = yes
					}
					check_variable = {
						var = PB_RUS_Rasputin_Influence_var
						value = 0.75
						compare = greater_than
					}
				}
				country_event = debug.32	# temporarily referring to the debug event until the rasputin events are made
			}
			else_if = {
				limit = {
					NOT = {
						has_country_leader = {
							character = RUS_Xenia_Olga
							ruling_only = yes
						}
					}
					check_variable = {
						var = PB_RUS_Rasputin_Influence_var
						value = 0.75
						compare = greater_than
					}
				}
				country_event = debug.32
			}
		}
	}
}
