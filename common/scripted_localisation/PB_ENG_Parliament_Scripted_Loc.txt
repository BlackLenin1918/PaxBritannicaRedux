defined_text = { #OPPOSITION SUPPORT
    name = get_opposition_support
    
    text = { #LOW OPPOSITION SUPPORT
        trigger = {
            check_variable = { opposition_support < 40 }
			NOT = {
				has_country_flag = ENG_Parliament_Suspended
			}
        }
        localization_key = "ENG_Parliament_Opposition_Support_Low"
    }

    text = { #MEDIUM OPPOSITION SUPPORT
        trigger = {
            check_variable = { opposition_support > 40 }
            check_variable = { opposition_support < 50 }
			NOT = {
				has_country_flag = ENG_Parliament_Suspended
			}
        }
        localization_key = "ENG_Parliament_Opposition_Support_Medium"
    }

    text = { #HIGH OPPOSITION SUPPORT
        trigger = {
            check_variable = { opposition_support > 50 }
			NOT = {
				has_country_flag = ENG_Parliament_Suspended
			}
        }
        localization_key = "ENG_Parliament_Opposition_Support_High"
    }

    text = { #PARLIAMENT SUSPENDED
        trigger = {
            has_country_flag = ENG_Parliament_Suspended
        }
        localization_key = "ENG_Parliament_Suspended"
    }
}
