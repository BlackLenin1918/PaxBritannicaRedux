defined_text = { #Imperial Spheres - Quebec
    name = get_imperial_sphere_quebec
    
    text = { 
        trigger = {
            QUE = {

            }
        }
        localization_key = "PB_Quebec_Pro_American"
    }

    text = { 
        trigger = {
