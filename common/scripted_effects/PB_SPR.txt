#Add to scripted triggers if more levels of hyper inflation is added

#Important to put HIGHEST first

SPR_unemployment_level_down = {
custom_effect_tooltip = SPR_unemployment_down
	if = {
		limit = { has_idea = PB_SPR_unemployment_issue_3 }
		#Makes the tooltip clearer if player looks at the effect early in the game
		hidden_effect = { remove_ideas = PB_SPR_unemployment_issue_3 }
	}
	if = {
		limit = { has_idea = PB_SPR_unemployment_issue_2 }
		swap_ideas = {
			remove_idea = PB_SPR_unemployment_issue_2
			add_idea = PB_SPR_unemployment_issue_3 
		}
	}
	if = {
		limit = { has_idea = PB_SPR_unemployment_issue_1 }
		swap_ideas = {
			remove_idea = PB_SPR_unemployment_issue_1
			add_idea = PB_SPR_unemployment_issue_2
		}
	}
	if = {
		limit = { has_idea = PB_SPR_unemployment_issue}
		swap_ideas = {
			remove_idea = PB_SPR_unemployment_issue
			add_idea = PB_SPR_unemployment_issue_1
		}
	}

}
