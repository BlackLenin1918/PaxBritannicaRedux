#Add to scripted triggers if more levels of hyper inflation is added

#Important to put HIGHEST first

PB_increase_centralization_small = {
custom_effect_tooltip = centralization_increase_five_tt
	if = {
		limit = { 
			check_variable = { centralization < 1 }
			OR = {
				has_government = jc
				has_government = ds
				has_government = ni
			}
		}
		add_to_variable = {
			var = centralization
			value = 0.05
		}
	}

	if = {
		limit = { 
			check_variable = { centralization < 0.8 }
			has_government = rc
		}
		add_to_variable = {
			var = centralization
			value = 0.05
		}
	}

	if = {
		limit = { 
			check_variable = { centralization < 0.6 }
			has_government = ss
		}
		add_to_variable = {
			var = centralization
			value = 0.05
		}
	}

	if = {
		limit = { 
			check_variable = { centralization < 0.5 }
			OR = {
				has_government = cn
				has_government = lb
			}
		}
		add_to_variable = {
			var = centralization
			value = 0.05
		}
	}

	if = {
		limit = { 
			check_variable = { centralization < 0.3 }
			has_government = an
		}
		add_to_variable = {
			var = centralization
			value = 0.05
		}
	}
}
