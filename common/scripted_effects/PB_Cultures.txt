## Changes the culture of the state scope into the culture specified in `temp_culture`
# SCOPE: State
PB_Cultures_change_culture_of_state = {
	log = "PB_Cultures_change_culture_of_state - [THIS.GetName] - New:[get_culture_name] Old:[get_main_culture_name]"
	set_variable = { culture_token = temp_culture }
	every_country = {
		limit = {
			is_ai = no
		}
		force_update_map_mode = { mapmode = culture_map_mode }
	}
	custom_effect_tooltip = PB_Cultures_change_culture_of_state_tt
}

## Adds the culture specified in `temp_culture` into the scoped country's list of accepted cultures
# SCOPE: Country
PB_Cultures_add_accepted_culture = {
	log = "PB_Cultures_add_accepted_culture - [THIS.GetName] - [get_culture_name]"
	add_to_array = { PB_accepted_cultures = temp_culture }
	custom_effect_tooltip = PB_Cultures_add_accepted_culture_tt
	#add_to_variable = { culture_dirty_var = 1 }
}

## Removes the culture specified in `temp_culture` from the scoped country's list of accepted cultures
# SCOPE: Country
PB_Cultures_remove_accepted_culture = {
	log = "PB_Cultures_remove_accepted_culture - [THIS.GetName] - [get_culture_name]"
	remove_from_array = { PB_accepted_cultures = temp_culture }
	custom_effect_tooltip = PB_Cultures_remove_accepted_culture_tt
	#add_to_variable = { culture_dirty_var = 1 }
}

## Sets the culture specified in `temp_culture` as the scoped country's main culture
# SCOPE: Country
PB_Cultures_set_main_culture = {
	log = "PB_Cultures_set_main_culture - [THIS.GetName] - New:[get_culture_name] Old:[get_main_culture_name]"
	set_variable = { PB_main_culture = temp_culture }
	if = {
		limit = {
			is_in_array = { PB_accepted_cultures = temp_culture }
		}
		remove_from_array = { PB_accepted_cultures = temp_culture }
		add_to_array = { array = PB_accepted_cultures value = temp_culture index = 0 }
	}
	custom_effect_tooltip = PB_Cultures_set_main_culture_tt
	#add_to_variable = { culture_dirty_var = 1 }
}
