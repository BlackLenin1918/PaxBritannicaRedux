PB_OBW_Calculate_Fail_Chance = {
	hidden_effect = {
		if = {
			limit = {
				check_variable = { Local_Launch_Failure_Chance > Local_Launch_Success_Chance }
			}
			set_country_flag = OBW_Critical_Failure_Assured
		}

		if = {
			limit = {
				OR = {
					check_variable = { Local_Launch_Failure_Chance > 75 }
					check_variable = { Local_Launch_Failure_Chance = 75 }
				}
			}
			set_country_flag = OBW_Critical_Failure_High
		}

		if = {
			limit = {
				OR = {
					check_variable = { Local_Launch_Failure_Chance > 50 }
					check_variable = { Local_Launch_Failure_Chance = 50 }
				}
			}
			set_country_flag = OBW_Critical_Failure_Moderate
		}

		if = {
			limit = {
				OR = {
					check_variable = { Local_Launch_Failure_Chance > 25 }
					check_variable = { Local_Launch_Failure_Chance = 25 }
				}
			}
			set_country_flag = OBW_Critical_Failure_Low
		}

		if = {
			limit = {
				OR = {
					check_variable = { Local_Launch_Failure_Chance < 25 }
				}
			}
			set_country_flag = OBW_Critical_Failure_Very_Low
		}
	}
}

PB_OBW_Reset_Fail_Chance = {
	hidden_effect = {
		clr_country_flag = OBW_Critical_Failure_Assured
		clr_country_flag = OBW_Critical_Failure_High
		clr_country_flag = OBW_Critical_Failure_Moderate
		clr_country_flag = OBW_Critical_Failure_Low
		clr_country_flag = OBW_Critical_Failure_Very_Low
	}
}
